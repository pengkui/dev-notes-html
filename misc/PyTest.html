/*
 * Convert an IPv4 dotted string to its 32-bit int
 * Python equivalent: ipstr2int = \
 * lambda ipstr: reduce(lambda x, y: (x << 8) | y, map(int, ipstr.split('.')))
 */
long ipv4_dotstr_to_int (const char * ipdotstr)
{
    long iplong = 0;
    char *_ipdotstr = strdup (ipdotstr);
    char ** quads = split_(_ipdotstr, ".", 4);
    int i;
    for (i=0; i<4; i++) {
        iplong = (iplong << 8) | atoi(quads[i]);
    }
    free (_ipdotstr);
    free (quads);
    return iplong;
}


/* Whether a given string is an IPv4 address. */
// http://stackoverflow.com/questions/791982/determine-if-a-string-is-a-valid-ip-address-in-c
int is_valid_ipv4_addr (const char * ipstr)
{
    struct sockaddr_in sa;
    int n = inet_pton (AF_INET, ipstr, &(sa.sin_addr));
    return n !=	0; /* 1 or 0 */
}
